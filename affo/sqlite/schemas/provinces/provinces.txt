CREATE TABLE IF NOT EXISTS 'provinces' 
('userId' INTEGER NOT NULL,
'id' INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
'provinceName' VARCHAR(240),
'cityCount' INTEGER NOT NULL DEFAULT 1,
'land' INTEGER NOT NULL DEFAULT 1,
'population' INTEGER NOT NULL DEFAULT 0,
'energy' INTEGER NOT NULL DEFAULT 0,
'happiness' INTEGER NOT NULL DEFAULT 0,
'pollution' INTEGER NOT NULL DEFAULT 0,
'productivity' INTEGER NOT NULL DEFAULT 0,
'consumer_spending' INTEGER NOT NULL DEFAULT 0);
