CREATE TABLE military
(id INTEGER PRIMARY KEY NOT NULL,
soldiers INTEGER NOT NULL DEFAULT 0,
artillery INTEGER NOT NULL DEFAULT 0,
tanks INTEGER NOT NULL DEFAULT 0,
bombers INTEGER NOT NULL DEFAULT 0,
fighters INTEGER NOT NULL DEFAULT 0,
apaches INTEGER NOT NULL DEFAULT 0,
spies INTEGER NOT NULL DEFAULT 0,
ICBMs INTEGER NOT NULL DEFAULT 0,
nukes INTEGER NOT NULL DEFAULT 0,
destroyers INTEGER NOT NULL DEFAULT 0,
cruisers INTEGER NOT NULL DEFAULT 0,
submarines INTEGER NOT NULL DEFAULT 0,
default_defense TEXT NOT NULL DEFAULT 'soldiers,tanks,artillery',
defcon INTEGER NOT NULL DEFAULT 1);
;
