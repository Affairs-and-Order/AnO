CREATE TABLE military
(id INTEGER PRIMARY KEY NOT NULL,
soldiers BIGINT NOT NULL DEFAULT 0,
artillery BIGINT NOT NULL DEFAULT 0,
tanks BIGINT NOT NULL DEFAULT 0,
bombers BIGINT NOT NULL DEFAULT 0,
fighters BIGINT NOT NULL DEFAULT 0,
apaches BIGINT NOT NULL DEFAULT 0,
spies BIGINT NOT NULL DEFAULT 0,
ICBMs BIGINT NOT NULL DEFAULT 0,
nukes BIGINT NOT NULL DEFAULT 0,
destroyers BIGINT NOT NULL DEFAULT 0,
cruisers BIGINT NOT NULL DEFAULT 0,
submarines BIGINT NOT NULL DEFAULT 0,
default_defense TEXT NOT NULL DEFAULT 'soldiers,tanks,artillery',
defcon BIGINT NOT NULL DEFAULT 1);
